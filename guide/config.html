# 结构

以下讲解全部以 TS 为例

// .vuepress/config.ts
import { defineUserConfig } from "vuepress";
import smplayer from "vuepress-plugin-smplayer";

export default defineUserConfig({
  plugins: [
    smplayer({
      bilibili: {
        danmaku: true,
        width: "100%",
        ratio: 16 / 9,
      },
      xigua: {
        autoplay: false,
        width: "100%",
        ratio: 16 / 9,
      },
      artplayer: {
        width: "100%",
        ratio: 16 / 9,
        src: {
          fullscreen: true,
          autoSize: true,
          setting: true,
          whitelist: ["*"],
          //所有 artplayer 配置项都可以配置
        },
      },
    }),
  ],
});
